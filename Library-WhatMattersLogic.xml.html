parameter &quot;Follow-Up Interval&quot; Interval&lt;System.Quantity&gt;
  default Interval[2 weeks, 26 weeks]

context Patient

define &quot;What Matters Observations&quot;:
  [Observation] obs
    where obs.category in &quot;What Matters Categories&quot;
      or obs.code in &quot;Well-Being Domains&quot;

// Returns a list of what matters Observations this Goal addresses.
define fluent function whatMattersObservations(goal Goal):
  flatten( goal.addresses GA
      return [Observation] obs
        where obs.id = GA.reference.getId()
          and obs.isWhatMatters()
  )

define fluent function isWhatMatters(obs Observation):
  &quot;What Matters Observations&quot; contains obs
